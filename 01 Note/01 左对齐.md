## 例子 1

$$
\begin{align}
          y = x \\
          y = 3x^2 + 5x + 2
          \end{align}
$$

~~~
\begin{align}
          y = x \\
          y = 3x^2 + 5x + 2
          \end{align}
~~~



## 例子 2 —— 用`&`字符调整对齐位置

$$
\begin{align}
          y & = x \\
          y & = 3x^2 + 5x + 2
          \end{align}
$$

~~~
\begin{align}
          y & = x \\
          y & = 3x^2 + 5x + 2
          \end{align}
~~~
